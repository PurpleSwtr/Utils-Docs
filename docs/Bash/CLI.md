```
touch <имя_файла> - создать файл
mkfir <имя папки> - создать папку
ls -la - отображение списка файлов с подробной информацией
echo 'str' - вывести в консоль
echo 'str' > example.txt - вывести в файл
-n - без переноса строки
head - показывает строки в файле
< > - перенаправление куда-либо, откуда либо в файл / из файла
<< >> - с добавлением строчек
```
